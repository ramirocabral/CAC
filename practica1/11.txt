ORG 1000H

CADENA DB "TheGame"
F DB 0
CAR DB "e"
ORG 3000h

CONCAR: MOV AL, [BX]
        CMP AL, 0   ; SI AL == 0
        JZ FIN
        CMP AL,AH ; SI AL == CAR
        JNZ SEGUIR
        MOV BYTE PTR [BX], 88  ; POSICION == "X"
        INC DH 
SEGUIR: INC DL
        INC BX  ; PASAMOS AL SIGUIENTE CARACTER
        JMP CONCAR
FIN:    RET
        
ORG 2000H 

MOV BX, OFFSET CADENA
MOV DL, 0 ; DL = CANTIDAD DE CARACTERES DEL STRING
MOV DH,0  ; DH = CANTIDAD DE VECES QUE APARECE CAR

MOV AH, CAR
CALL CONCAR

HLT

END
